154static void ui_fill_lcd_buf_program(ui_program_t *program, uint8_t program_index,
155 char text_buf_top[][17], char text_buf_bottom[][17], uint8_t length)
156{
157 char gradient_buf[5];
158 char sign;
159
160 for (int i = 0; i < length; ++i) {
161 sign = (program->gradient_negative[i] == 1) ? '-' : ' ';
162 uint16_t gradient = program->gradient[i];
163 if(100 > gradient)
164 {
165 snprintf(gradient_buf, 5, " %c%2d", sign, gradient);
166 }
167 else
168 {
169 snprintf(gradient_buf, 5, "%c%3d", sign, gradient);
170 }
171 snprintf(text_buf_top[i], sizeof(text_buf_top[i]), " P%d: %d/%d ",
172 program_index+1, i+1, length);
173 snprintf(text_buf_bottom[i], sizeof(text_buf_bottom[i]), "%s -> %4d C",
174 gradient_buf, program->temperature[i]);
175 }
176}

189static void ui_fill_program_values(ui_program_t *program)
190{
191 temp_sign = (temp_counter == 0 && temp_sign == 1) ? 0 : temp_sign;
192 switch (scroll_counter % 2)
193 {
194 case 0: /* Temperature or length element */
195 /* Limit to positive values only */
196 if(temp_sign == 1)
197 {
198 temp_sign = 0;
199 temp_counter = 0;
200 }
201 /* Length element (scroll_counter == 0) */
202 if (0 == scroll_counter) {
203 if(MAX_PROGRAM_SEQ_LENGTH < temp_counter_single)
204 {
205 temp_counter_single = MAX_PROGRAM_SEQ_LENGTH;
206 }
207 if(MIN_PROGRAM_SEQ_LENGTH > temp_counter_single)
208 {
209 temp_counter_single = MIN_PROGRAM_SEQ_LENGTH;
210 }
211 program->length = temp_counter_single;
212 break;
213 }
214 /* Temperature element */
215 if(MAX_TEMPERATURE < temp_counter)
216 {
217 temp_counter = MAX_TEMPERATURE;
218 }
219 program->temperature[(scroll_counter-1)/2] = temp_counter;
220 break;
221
222 case 1: /* Gradient element */
223 if(MAX_GRADIENT < temp_counter)
224 {
225 temp_counter = MAX_GRADIENT;
226 }
227 program->gradient[(scroll_counter-1)/2] = temp_counter;
228 program->gradient_negative[(scroll_counter-1)/2] = temp_sign;
229 break;
230
231 default:
232 break;
233 }
234}

247static HAL_StatusTypeDef ui_update_create_program_detailed(Ui_HandleTypeDef_t *ui, event_type_t event)
248{
249 uint8_t index = ui->programs.length;
250 uint8_t length = c_program.length;
251
252 switch (event) {
253 case NO_EVENT:
254 break;
255
256 case BUT1:
257 case ENC_DOWN: /* Decrease value */
258 if(0 >= temp_counter || temp_sign == 1)
259 {
260 temp_sign = 1;
261 temp_counter = (event == BUT1) ? (temp_counter + BUTTON_INC) : (temp_counter + ENC_INC);
262 temp_counter_single++;
263 }
264 else
265 {
266 if(event == BUT1)
267 {
268 temp_counter = (BUTTON_INC > temp_counter) ? 0 : (temp_counter - BUTTON_INC);
269 }
270 else
271 {
272 temp_counter = (ENC_INC > temp_counter) ? 0 : (temp_counter - ENC_INC);
273 }
274 temp_counter_single--;
275 }
276 break;
277
278 case BUT2:
279 case ENC_UP: /* Increase value */
280 if(0 >= temp_counter || temp_sign == 0)
281 {
282 temp_sign = 0;
283 temp_counter = (event == BUT2) ? (temp_counter + BUTTON_INC) : (temp_counter + ENC_INC);
284 temp_counter_single++;
285 }
286 else
287 {
288 if(event == BUT2){
289 temp_counter = (BUTTON_INC > temp_counter) ? 0 : (temp_counter - BUTTON_INC);
290 }
291 else
292 {
293 temp_counter = (ENC_INC > temp_counter) ? 0 : (temp_counter - ENC_INC);
294 }
295 temp_counter_single--;
296 }
297 break;
298
299 case BUT3: /* Navigate back - cancel program creation */
300 ui->state = CREATE_PROGRAM;
301 c_program.length = 1;
302 c_program.gradient[0] = 0;
303 c_program.gradient_negative[0] = 0;
304 c_program.temperature[0] = 0;
305 scroll_counter = 0;
306 temp_counter_single = 0;
307 break;
308
309 case BUT4: /* Start/stop - not implemented */
310 break;
311
312 case ENC_BUT: /* Enter - advance to next field or save */
313 if((length*2) <= scroll_counter)
314 {
315 /* Save program */
316 ui->programs.length++;
317 ui->programs.program_list[index] = c_program;
318 ui->programs.cur_index = index;
319 /* Reset temporary values */
320 c_program.length = 1;
321 c_program.gradient[0] = 0;
322 c_program.gradient_negative[0] = 0;
323 c_program.temperature[0] = 0;
324 scroll_counter = 0;
325 temp_counter = 0;
326 temp_counter_single = 0;
327 ui->state = PROGRAMS_OVERVIEW;
328 break;
329 }
330 scroll_counter++;
331 temp_counter = 0;
332 temp_counter_single = 0;
333 break;
334
335 default:
336#ifdef UI_ENABLE_LOG
337 logMsg(LOG_ERROR, "unknown event: %u", event);
338#endif
339 return HAL_ERROR;
340 }
341
342 ui_fill_program_values(&c_program);
343
344 char text_buf_top[length][UI_LCD_CHAR_SIZE];
345 char text_buf_bottom[length][UI_LCD_CHAR_SIZE];
346 ui_fill_lcd_buf_program(&c_program, index, text_buf_top, text_buf_bottom, length);
347
348 if(scroll_counter == 0)
349 {
350 char text_buf[20];
351 snprintf(text_buf, sizeof(text_buf), " length: %2d ", c_program.length);
352 ui_print_lcd(ui, text_buf, empty_text_buf);
353 return HAL_OK;
354 }
355
356 ui_print_lcd(ui, text_buf_top[(scroll_counter-1)/2], text_buf_bottom[(scroll_counter-1)/2]);
357 ui_print_lcd_program_symb(ui);
358
359 return HAL_OK;
360}

559static HAL_StatusTypeDef ui_update_settings_overview(Ui_HandleTypeDef_t *ui, event_type_t event)
560{
561 uint8_t index;
562
563 switch (event) {
564 case NO_EVENT:
565 break;
566
567 case BUT1:
568 case ENC_DOWN:
569 if(temp_bool) /* Edit mode: adjust value */
570 {
571 index = ui->settings.cur_index;
572 float32_t val = ui->settings.setting_list[index].value;
573 float32_t inc_millis = (event==BUT1) ? BUTTON_INC_FLOAT_MILLIS : ENC_INC_FLOAT_MILLIS;
574 val -= (inc_millis / 1000);
575 if (val > MAX_SETTING) {
576 val = MAX_SETTING;
577 } else if (val < -MAX_SETTING) {
578 val = -MAX_SETTING;
579 }
580 ui->settings.setting_list[index].value = val;
581 break;
582 }
583 /* Navigation mode: scroll list */
584 ui->settings.cur_index--;
585 if(ui->settings.cur_index >= ui->settings.length)
586 {
587 ui->settings.cur_index = ui->settings.length-1;
588 }
589 break;
590
591 case BUT2:
592 case ENC_UP:
593 if(temp_bool) /* Edit mode: adjust value */
594 {
595 index = ui->settings.cur_index;
596 float32_t val = ui->settings.setting_list[index].value;
597 float32_t inc_millis = (event==BUT2) ? BUTTON_INC_FLOAT_MILLIS : ENC_INC_FLOAT_MILLIS;
598 val += (inc_millis / 1000);
599 if (val > MAX_SETTING) {
600 val = MAX_SETTING;
601 } else if (val < -MAX_SETTING) {
602 val = -MAX_SETTING;
603 }
604 ui->settings.setting_list[index].value = val;
605 break;
606 }
607 /* Navigation mode: scroll list */
608 ui->settings.cur_index++;
609 if(ui->settings.cur_index >= ui->settings.length)
610 {
611 ui->settings.cur_index = 0;
612 }
613 break;
614
615 case BUT3:
616 ui->settings.cur_index = 0;
617 ui->state = SETTINGS;
618 break;
619
620 case BUT4:
621 break;
622
623 case ENC_BUT: /* Toggle edit mode */
624 temp_bool = (temp_bool) ? 0 : 1;
625 break;
626
627 default:
628#ifdef UI_ENABLE_LOG
629 logMsg(LOG_ERROR, "unknown event: %u", event);
630#endif
631 return HAL_ERROR;
632 }
633
634 index = ui->settings.cur_index;
635 ui_setting_t setting = ui->settings.setting_list[index];
636 char buf[UI_LCD_CHAR_SIZE];
637 snprintf(buf, sizeof(buf), " %.1f ", setting.value);
638
639 ui_print_lcd(ui, setting.name, buf);
640
641 return HAL_OK;
642}

798HAL_StatusTypeDef ui_update(Ui_HandleTypeDef_t *ui)
799{
800 event_type_t cur_event = ui_get_events(ui);
801
802 /* Skip update if no change */
803 if(ui->last_state == ui->state && cur_event == NO_EVENT)
804 {
805 return HAL_OK;
806 }
807 ui->last_state = ui->state;
808
809 switch (ui->state) {
810 case SETTINGS:
811 return ui_update_settings(ui, cur_event);
812
813 case SETTINGS_OVERVIEW:
814 return ui_update_settings_overview(ui, cur_event);
815
816 case SETPOINT:
817 return ui_update_setpoint(ui, cur_event);
818
819 case PROGRAMS:
820 return ui_update_programs(ui, cur_event);
821
822 case PROGRAMS_OVERVIEW:
823 return ui_update_programs_overview(ui, cur_event);
824
825 case PROGRAM_DETAILED:
826 return ui_update_program_detailed(ui, cur_event);
827
828 case CREATE_PROGRAM:
829 return ui_update_create_program(ui, cur_event);
830
831 case CREATE_PROGRAM_DETAILED:
832 return ui_update_create_program_detailed(ui, cur_event);
833
834 default:
835 return HAL_OK;
836 }
837 return HAL_OK;
838}
