116static void heater_update_pwm_coil(heater_coil_t* coil)
117{
118 /* First call: initialize timestamp */
119 if(0 == coil->time_pwm_last){
120 coil->time_pwm_last = HAL_GetTick();

179static HAL_StatusTypeDef heater_check_door_state(Heater_HandleTypeDef_t* hheater)
180{
181 if(NULL == hheater){
182 return HAL_ERROR;
183 }
184
185 /* Door is open - pause heating */
186 if(hheater->flag_door_open)
187 {
188 /* Save previous level if not already saved */
189 if(0xff == hheater->heater_level_prev)
190 {
191 hheater->heater_level_prev = hheater->heater_level;
192 }
193
194 /* Turn off heater if not already off */
195 if(0 != hheater->heater_level)
196 {
197 if(HAL_OK != heater_set_level(hheater, 0))
198 {
199 return HAL_ERROR;
200 }
201 }
202 }
203
204 /* Door is closed */
205 if(!hheater->flag_door_open)
206 {
207 /* Resume previous level if one was saved */
208 if(0xff != hheater->heater_level_prev)
209 {
210 if(HAL_OK != heater_set_level(hheater, hheater->heater_level_prev))
211 {
212 return HAL_ERROR;
213 }
214 hheater->heater_level_prev = 0xff;
215 }
216 }
217 return HAL_OK;
218}

331float32_t heater_calculate_slope(Heater_HandleTypeDef_t* hheater) {
332
333 float32_t* temp = hheater->temperature;
334 float32_t mean_temp = 0;
335 uint32_t total_intervals = PID_CALC_INTERVAL_SECONDS / TEMPERATURE_SAMPLING_INTERVAL_SECONDS;
336 float32_t mean_time = ((total_intervals + 1.0f) * TEMPERATURE_SAMPLING_INTERVAL_SECONDS) / 2.0f;
337
338 arm_mean_f32(temp, total_intervals, &mean_temp);
339
340 /* Calculate numerator and denominator of slope formula */
341 float32_t numerator = 0;
342 float32_t denominator = 0;
343 for (uint32_t i = 0; i < total_intervals; i++) {
344 numerator += (i * TEMPERATURE_SAMPLING_INTERVAL_SECONDS - mean_time) * (temp[i] - mean_temp);
345 denominator += (i * TEMPERATURE_SAMPLING_INTERVAL_SECONDS - mean_time)
346 * (i * TEMPERATURE_SAMPLING_INTERVAL_SECONDS - mean_time);
347 }
348 float32_t slope = numerator / denominator;
349
350 return slope;
351}
