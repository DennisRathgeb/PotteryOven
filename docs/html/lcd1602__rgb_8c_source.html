46HAL_StatusTypeDef lcd1602_begin(LCD1602_RGB_HandleTypeDef_t *hlcd, uint8_t cols, uint8_t lines)
47{
48 uint8_t err_c = HAL_OK;
49 if (lines > 1) {
50 hlcd->showfunction |= LCD_2LINE;
51 }
52 hlcd->numlines = lines;
53 hlcd->currline = 0;
54
55#ifdef LCD_USE_RTOS_INIT
56 osDelay(50);
57#else
58 HAL_Delay(50);
59#endif
60
61 err_c += lcd1602_command(hlcd, LCD_FUNCTIONSET | hlcd->showfunction);
62
63#ifdef LCD_USE_RTOS_INIT
64 osDelay(5);
65#else
66 HAL_Delay(5);
67#endif
68
69 err_c += lcd1602_command(hlcd, LCD_FUNCTIONSET | hlcd->showfunction);
70#ifdef LCD_USE_RTOS_INIT
71 osDelay(5);
72#else
73 HAL_Delay(5);
74#endif
75
76 err_c += lcd1602_command(hlcd, LCD_FUNCTIONSET | hlcd->showfunction);
77
78 hlcd->showcontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
79 err_c += lcd1602_display(hlcd);
80
81 err_c += lcd1602_clear(hlcd);
82
83 hlcd->showmode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
84 err_c += lcd1602_command(hlcd, LCD_ENTRYMODESET | hlcd->showmode);
85
86 err_c += lcd1602_setReg(hlcd, REG_MODE1, 0);
87 err_c += lcd1602_setReg(hlcd, REG_OUTPUT, 0xFF);
88 err_c += lcd1602_setReg(hlcd, REG_MODE2, 0x20);
89
90 err_c += lcd1602_setColorWhite(hlcd);
91 if(err_c)
92 {
93 return err_c;
94 }
95 return HAL_OK;
96}

415HAL_StatusTypeDef lcd1602_customSymbol(LCD1602_RGB_HandleTypeDef_t *hlcd,
416 uint8_t location, uint8_t charmap[])
417{
418 uint8_t err_c = HAL_OK;
419 location &= 0x7;
420 err_c += lcd1602_command(hlcd, LCD_SETCGRAMADDR | (location << 3));
421
422 uint8_t data[9];
423 data[0] = 0x40;
424 for(int i=0; i<8; i++)
425 data[i+1] = charmap[i];
426
427 err_c += lcd1602_send(hlcd, data, 9);
428 if (err_c)
429 {
430 return err_c;
431 }
432 return HAL_OK;
433}
