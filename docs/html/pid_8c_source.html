87PID_Controller_Output_t PID_CalculateOutput(PID_HandletypeDef_t *hpid,
88 float32_t current_temperature,
89 float32_t setpoint) {
90 float32_t error = setpoint - current_temperature;
91 integral += error;
92 float32_t derivative = error - last_error;
93
94 /* Apply first-order low-pass filter to derivative term */
95 float32_t filtered_derivative = (1.0f - hpid->derivative_filter_coeff) * last_derivative +
96 hpid->derivative_filter_coeff * derivative;
97 last_derivative = filtered_derivative;
98 last_error = error;
99
100 /* Calculate PID output */
101 float32_t output = hpid->k_proportional * error +
102 hpid->k_integral * integral +
103 hpid->k_derivative * derivative;
104
105 /* Apply hysteresis for bang-bang control */
106 if (output > hpid->hysteresis) {
107 return PID_OUTPUT_ON;
108 } else if (output < -hpid->hysteresis) {
109 return PID_OUTPUT_OFF;
110 } else {
111 return PID_OUTPUT_OFF; /* Default: stay off */
